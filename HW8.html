corner_points = np.array([[0, 3.5], [2, 3], [4, 2], [6, 0]])

def F(x1, x2):
    return 4*x1 + 6*x2

max_value = float('-inf')
max_point = None

for point in corner_points:
    value = F(point[0], point[1])
    if value > max_value:
        max_value = value
        max_point = point

print(f"Max value: {max_value} in point {[coord for coord in max_point]}")
